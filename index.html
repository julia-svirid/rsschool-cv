Get max element using rest operator

                        
 const getMax = (arr) => { 
     if (arr.length === 0) return null;
     let [max, ...rest] = arr; 
     for (let i = 0; i < rest.length; i++) { 
         if (max < rest[i]) {
             max = rest[i]; 
         } 
     } 
     return max; 
 };
